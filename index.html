Welcome to my index.html page!

const express = require('express');
const app = express();
const port = 3000;
app.get('/', function(req, res) {
res.sendFile(__dirname + "/index.html", function(error) {
if (error) {
console.log('There was an error retreving about.html');
} else {
console.log('Was able to send about.html');
}
});
});
app.get('/about', function(req, res) {
res.sendFile(__dirname + "/about.html", function(error) {
if (error) {
console.log('There was an error retreving about.html');
} else {
console.log('Was able to send about.html');
}
});
});
app.listen(port, () => console.log(`server is running on port ${port}`));
